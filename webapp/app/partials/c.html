#include <stdio.h>

// The Fibonacci Function
void print_fibonacci(int num) {
int i, j, k, cnt;
      i = 0;
      j = 1;
      cnt = 2;
      printf("%d %d ", i, j);
      k = i + j;
      while (cnt < num) {
            printf("%d ", k);
            i = j;
            j = k;
            k = i + j;
            cnt++;
      }
      printf("\n");
      return;
}
