#include <iostream>

// The Fibonacci Function
class fib
{
public:
      int a, b, c;
      void generator(int);
};

void fib::generator(int n) {
      a = 0;
      b = 1;
      cout << a << " " << b;
      for (int i = 1; i <= n-2; i++) {
            c = a + b;
            cout << " " << c;
            a = b;
            b = c;
      }
}

int main (void){
      int n;
      cout << "Enter number of terms you need in the series: ";
      cout >> n;
      fib ob;
      ob.generator(n);
}
